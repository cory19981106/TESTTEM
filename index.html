// 修改後的後台渲染邏輯 function renderOrders(){ const s = document.getElementById('o-search').value.toLowerCase(); const start = document.getElementById('o-start').value; const end = document.getElementById('o-end').value; let filtered = orders.filter(o => { const ms = (o.name || "").includes(s) || (o.phone || "").includes(s); let md = true; if(start && o.createdAt) md = o.createdAt.toDate() >= new Date(start); if(end && o.createdAt) { let e = new Date(end); e.setHours(23,59,59); md = md && o.createdAt.toDate() <= e; } return ms && md; }); let html = ""; filtered.forEach(o => { const c = o.status==='已查帳'; // 組合信眾資訊字串 const birthInfo = o.birthday ? `
生辰：${o.birthday} (${o.birthHour})` : ""; html += ` ${o.createdAt?o.createdAt.toDate().toLocaleDateString():'-'} ${o.category} ${o.name}
${o.phone}${birthInfo} ${o.item} $${o.amount} `; }); document.getElementById('order-rows').innerHTML = html; } // 修改後的 Excel 匯出邏輯 function exportExcel(){ const exportData = orders.map(o => ({ "登記日期": o.createdAt ? o.createdAt.toDate().toLocaleString() : "", "類別": o.category, "姓名": o.name, "電話": o.phone, "出生日期": o.birthday || "無", "出生時辰": o.birthHour || "無", "項目": o.item, "金額": o.amount, "狀態": o.status })); const ws = XLSX.utils.json_to_sheet(exportData); const wb = XLSX.utils.book_new(); XLSX.utils.book_append_sheet(wb, ws, "廣寧宮財務報表"); XLSX.writeFile(wb, `員林廣寧宮報表_${new Date().toLocaleDateString()}.xlsx`); }
